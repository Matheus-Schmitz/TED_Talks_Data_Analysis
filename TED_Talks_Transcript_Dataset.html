def get_split(text1):
    l_total = []
    l_parcial = []
    n = (len(text1.split())//128)+1
    for w in range(n):
        l_parcial = text1.split()[w*128:w*128 + 128]
        l_total.append(" ".join(l_parcial))
    return l_total
